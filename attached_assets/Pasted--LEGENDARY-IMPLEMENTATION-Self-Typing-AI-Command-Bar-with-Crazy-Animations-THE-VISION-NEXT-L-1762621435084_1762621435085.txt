interface TypewriterState {
  displayText: string;
  currentPromptIndex: number;
  isTyping: boolean;
  isDeleting: boolean;
  isPaused: boolean;
}

export const SelfTypingCommandBar: React.FC<{
  onManualSubmit?: (query: string) => void;
}> = ({ onManualSubmit }) => {
  const [state, setState] = useState<TypewriterState>({
    displayText: '',
    currentPromptIndex: 0,
    isTyping: true,
    isDeleting: false,
    isPaused: false,
  });

  const [isHovered, setIsHovered] = useState(false);
  const [isFocused, setIsFocused] = useState(false);
  const [manualInput, setManualInput] = useState('');
  const [showCursor, setShowCursor] = useState(true);
  const inputRef = useRef<HTMLInputElement>(null);

  const currentPrompt = EPIC_PROMPTS[state.currentPromptIndex];

  // ğŸ¨ Typewriter Effect - The Magic Happens Here
  useEffect(() => {
    // Pause animation when user interacts
    if (isHovered || isFocused || manualInput) {
      return;
    }

    const typingSpeed = 80; // ms per character
    const deletingSpeed = 40; // faster deletion
    const pauseAfterTyping = 2500; // pause to read
    const pauseAfterDeleting = 500; // brief pause before next

    const timer = setTimeout(() => {
      const targetText = currentPrompt.text;

      // STATE: Typing
      if (state.isTyping && !state.isPaused) {
        if (state.displayText.length < targetText.length) {
          setState((prev) => ({
            ...prev,
            displayText: targetText.slice(0, prev.displayText.length + 1),
          }));
        } else {
          // Finished typing - pause
          setState((prev) => ({
            ...prev,
            isTyping: false,
            isPaused: true,
          }));
        }
      }
      // STATE: Paused after typing
      else if (state.isPaused && !state.isTyping && !state.isDeleting) {
        setState((prev) => ({
          ...prev,
          isPaused: false,
          isDeleting: true,
        }));
      }
      // STATE: Deleting
      else if (state.isDeleting && !state.isPaused) {
        if (state.displayText.length > 0) {
          setState((prev) => ({
            ...prev,
            displayText: prev.displayText.slice(0, -1),
          }));
        } else {
          // Finished deleting - move to next prompt
          setState((prev) => ({
            ...prev,
            isDeleting: false,
            isPaused: true,
            currentPromptIndex: (prev.currentPromptIndex + 1) % EPIC_PROMPTS.length,
          }));
        }
      }
      // STATE: Paused after deleting
      else if (state.isPaused && state.isDeleting) {
        setState((prev) => ({
          ...prev,
          isPaused: false,
          isDeleting: false,
          isTyping: true,
        }));
      }
    }, state.isTyping ? typingSpeed : state.isDeleting ? deletingSpeed : state.isPaused && !state.isDeleting ? pauseAfterTyping : pauseAfterDeleting);

    return () => clearTimeout(timer);
  }, [state, currentPrompt, isHovered, isFocused, manualInput]);

  // Cursor blink effect
  useEffect(() => {
    const cursorTimer = setInterval(() => {
      setShowCursor((prev) => !prev);
    }, 530);
    return () => clearInterval(cursorTimer);
  }, []);

  // Keyboard shortcut (âŒ˜K / Ctrl+K)
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault();
        inputRef.current?.focus();
      }
      // ESC to blur
      if (e.key === 'Escape') {
        inputRef.current?.blur();
        setManualInput('');
      }
    };
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, []);

  const handleManualSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (manualInput.trim()) {
      onManualSubmit?.(manualInput);
      setManualInput('');
    }
  };

  const handleFocus = () => {
    setIsFocused(true);
  };

  const handleBlur = () => {
    setIsFocused(false);
    if (!manualInput) {
      // Resume animation
    }
  };

  return (
    <div className="w-full max-w-5xl mx-auto px-4">
      {/* Main Command Bar */}
      <form onSubmit={handleManualSubmit}>
        <div
          className="relative group"
          onMouseEnter={() => setIsHovered(true)}
          onMouseLeave={() => setIsHovered(false)}
        >
          {/* Gradient Border Effect */}
          <div
            className={`absolute -inset-0.5 bg-gradient-to-r ${currentPrompt.color} rounded-2xl blur opacity-0 group-hover:opacity-30 transition-all duration-500 animate-gradient`}
          />

          {/* Main Container */}
          <div
            className={`relative bg-white border-2 rounded-2xl transition-all duration-500 ${
              isFocused
                ? 'border-black shadow-2xl shadow-black/20 scale-[1.02]'
                : isHovered
                ? 'border-gray-400 shadow-xl'

            }`}
          >
